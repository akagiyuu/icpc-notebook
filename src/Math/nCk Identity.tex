\[
    \sum_{k = 0}^{n} {(-1)}^k \binom{r}{k} = {(-1)}^n \binom{r - 1}{n}
\]
\[
    \sum_{k = 0}^{n} \binom{r + k}{k} = \binom{r + n + 1}{n}
\]
\[
    \sum_{j = 0}^{m} \binom{n + j}{n} = \binom{n + m + 1}{m}    
\]
\[
    \sum_{k = 0}^{n} \binom{r}{k} \binom{s}{n - k} = \binom{r + s}{n}
\]
